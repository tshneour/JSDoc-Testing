/**
 * Adds four numbers.
 * @param {number} a The first number.
 * @param {number} b The second number.
 * @param {number} c The third number.
 * @param {number} d The fourth number.
 * @returns {number} The sum of the two numbers.
 */
function addFour(a, b, c, d) {
    return a + b + c + d;
}
